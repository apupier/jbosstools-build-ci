# 1. match line starting with 'error ', case-insensitive
error /(?i)^error /
# 2. other error regexs
error /BUILD FAILED/

error /exec returned: 13/

# 3. warnings
warning /[Ww]arning/
warning /WARNING/
warning /.+\[WARNING\].+/
warning /Connection refused/
warning /No baseline version/
